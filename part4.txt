        const streamVertexShader = `
            uniform vec3 uStart;
            uniform vec3 uEnd;
            uniform vec3 uControl1;
            uniform vec3 uControl2;

            varying vec2 vUv;
            varying float vProgress;

            vec3 getBezier(float t) {
                float u = 1.0 - t;
                float tt = t * t;
                float uu = u * u;
                float uuu = uu * u;
                float ttt = tt * t;

                vec3 p = uuu * uStart;
                p += 3.0 * uu * t * uControl1;
                p += 3.0 * u * tt * uControl2;
                p += ttt * uEnd;
                return p;
            }

            vec3 getDerivative(float t) {
                float u = 1.0 - t;
                vec3 p = 3.0 * u * u * (uControl1 - uStart);
                p += 6.0 * u * t * (uControl2 - uControl1);
                p += 3.0 * t * t * (uEnd - uControl2);
                return p;
            }

            void main() {
                vUv = uv;
                float t = position.y + 0.5;
                vProgress = t;

                vec3 pos = getBezier(t);
                vec3 tangent = normalize(getDerivative(t));
                
                vec3 up = vec3(0.0, 1.0, 0.0);
                if (abs(dot(up, tangent)) > 0.99) {
                    up = vec3(1.0, 0.0, 0.0);
                }
                vec3 right = normalize(cross(up, tangent));
                vec3 normal = cross(tangent, right);

                vec3 finalPos = pos + right * position.x + normal * position.z;

              if (node.userData.labelEl.style.opacity !== "0") {
                node.userData.labelEl.style.opacity = "0";
              }
            } else {
              if (node.userData.labelEl.style.opacity !== "0.8") {
                node.userData.labelEl.style.opacity = "0.8";
                node.userData.labelEl.style.left = "0px";
                node.userData.labelEl.style.top = "0px";
              }
              const x = (_tempVector.x * 0.5 + 0.5) * window.innerWidth;
              const y = (_tempVector.y * -0.5 + 0.5) * window.innerHeight;
              const yOffset = 40 * finalScale;
              node.userData.labelEl.style.transform = `translate3d(${x}px, ${y + yOffset}px, 0) translate(-50%, -50%)`;
            }
          });

          coreMesh.position.lerp(coreMesh.userData.basePos, dt * 2.0);
          coreLight.position.copy(coreMesh.position);

          agents.forEach((agent) => {
            const d = agent.userData;
            const targetY =
              d.basePos.y + Math.sin(time * 0.5 + d.phaseOffset) * 2.0;
            agent.position.x = lerp(agent.position.x, d.basePos.x, dt * 2.0);
            agent.position.z = lerp(agent.position.z, d.basePos.z, dt * 2.0);
            agent.position.y = lerp(agent.position.y, targetY, dt * 2.0);

            // --- NEW ACTIVATION ANIMATION ---
            const targetActivation = d.isActive ? 1.0 : 0.0;
            const lerpFactor = 1.0 - Math.exp(-dt * 10.0);
            d.activationLevel +=
              (targetActivation - d.activationLevel) * lerpFactor;

            // Spectacular BOUNCE pop effect
            const pop =
              Math.sin(d.activationLevel * Math.PI) *
              Math.pow(Math.sin(d.activationLevel * Math.PI), 0.5) *
              1.5;

            // Adding a sharp rotation spin when activating
            agent.rotation.y += dt * (0.2 + pop * 10.0);

            if (d.auraMat) {
              d.auraMat.uniforms.uTime.value = time;
              d.auraMat.uniforms.uActivation.value = d.activationLevel;
            }

            if (d.isAnalytic) {
              d.analyticCore.material.uniforms.uTime.value = time;
              d.analyticCore.material.uniforms.uActivation.value =
                d.activationLevel;
              d.analyticCore.material.uniforms.uPulse.value = lerp(
                d.analyticCore.material.uniforms.uPulse.value,
                0.0,
                dt * 8.0,
              );

              const pulseFactor = 1.0 + d.activationLevel * 5.0 + pop;
              d.analyticRings.forEach((r, idx) => {
                r.rotation.y += dt * (0.5 + idx * 0.2) * pulseFactor * 2.0;
                r.rotation.z += dt * (0.3 + idx * 0.3) * pulseFactor * 2.0;
                r.scale.setScalar(
                  lerp(1.0, 1.4, d.activationLevel) + pop * 0.5,
                );
              });

              d.analyticCore.scale.setScalar(
                lerp(1.0, 1.5, d.activationLevel) + pop,
              );
              d.analyticCore.rotation.y += dt * (1.0 + d.activationLevel * 4.0);
            } else if (d.isAuditor) {
              d.auditorBeam.material.uniforms.uTime.value = time;
              d.auditorBeam.material.uniforms.uActivation.value =
                d.activationLevel;
              d.auditorDisk.material.uniforms.uTime.value = time;
              d.auditorDisk.material.uniforms.uActivation.value =
                d.activationLevel;

              d.auditorCore.rotation.y += dt * (0.5 + d.activationLevel * 6.0);

              const beamScale = lerp(0.8, 1.8, d.activationLevel) + pop;
              d.auditorBeam.scale.set(beamScale, 1.0, beamScale);
              d.auditorDisk.scale.set(beamScale, beamScale, 1.0);

              _tempEyeColor
                .copy(_colorAuditorBase)
                .lerp(_colorAuditorActive, d.activationLevel);
              d.auditorEye.material.color.copy(_tempEyeColor);
              d.auditorEye.scale.setScalar(1.0 + pop * 0.5);

              const orbitSpeed = 1.0 + d.activationLevel * 8.0;
              const orbitRadius = 1.5 + d.activationLevel * 1.2 + pop;
              d.satellites.forEach((satData, idx) => {
                satData.angle += dt * orbitSpeed;
                satData.mesh.position.x = Math.cos(satData.angle) * orbitRadius;
                satData.mesh.position.z = Math.sin(satData.angle) * orbitRadius;
                satData.mesh.position.y =
                  Math.sin(time * 5.0 + idx) * (0.5 + d.activationLevel);
                satData.mesh.rotation.x += dt * (1.0 + d.activationLevel * 4.0);
                satData.mesh.rotation.y += dt * (1.5 + d.activationLevel * 4.0);
                satData.mesh.material.color.copy(_tempEyeColor);
              });
            } else if (d.isCognitive) {
              d.cogCore.material.uniforms.uTime.value = time;
              d.cogCore.material.uniforms.uActivation.value = d.activationLevel;
              d.cogCore.rotation.y += dt * (0.3 + d.activationLevel * 4.0);
              d.cogCore.rotation.x += dt * (0.1 + d.activationLevel * 2.0);
              d.cogCore.scale.setScalar(
                lerp(1.0, 1.4, d.activationLevel) + pop,
              );

              const orbitSpeed = 0.5 + d.activationLevel * 6.0;
              const orbitRadius = 1.5 + d.activationLevel * 1.5 + pop;
              d.nodes.forEach((node, idx) => {
                const t =
                  time * node.speed * (1.0 + d.activationLevel * 2.0) +
                  node.offset;
                node.mesh.position.x = Math.sin(t) * Math.cos(t) * orbitRadius;
                node.mesh.position.y = Math.sin(t) * orbitRadius;
                node.mesh.position.z = Math.cos(t) * orbitRadius;
                node.mesh.scale.setScalar(1.0 + pop * 0.5);
                node.mesh.material.color
                  .copy(_colorCognitiveBase)
                  .lerp(_colorCognitiveActive, d.activationLevel);
              });
            } else if (d.isExecutor) {
              const spinSpeed = 1.0 + d.activationLevel * 20.0;
              d.execGroupCenter.rotation.y += dt * spinSpeed;
              d.execCoreTop.position.y =
                lerp(0.0, 1.0, d.activationLevel) + pop * 0.5;
              d.execCoreBottom.position.y =
                lerp(-2.5, -3.5, d.activationLevel) - pop * 0.5;
              d.execCoreTop.scale.setScalar(1.0 + pop);
              d.execCoreBottom.scale.setScalar(1.0 + pop);

              d.rings.forEach((r, idx) => {
                r.mesh.rotation.z += dt * spinSpeed * r.dir;
                r.mesh.scale.setScalar(
                  lerp(1.0, 1.6 + idx * 0.3, d.activationLevel) + pop,
                );
              });

              d.execCoreTop.material.emissiveIntensity = lerp(
                1.0,
                8.0,
                d.activationLevel,
              );
              d.execCoreBottom.material.emissiveIntensity = lerp(
                1.0,
                8.0,
                d.activationLevel,
              );
            } else if (d.isIntegration) {
              d.intCenter.scale.setScalar(lerp(1.0, 1.8, d.activationLevel));

              const knotSpeed = 0.5 + d.activationLevel * 4.0;
              d.knots.forEach((k) => {
                k.mesh.rotateOnAxis(k.axis, dt * knotSpeed);
                k.mesh.scale.setScalar(lerp(1.0, 1.5, d.activationLevel));
                k.mesh.material.opacity = lerp(0.6, 1.0, d.activationLevel);
              });
            } else if (d.isSecurity) {
              d.secCore.rotation.y += dt * 0.2;
              d.secCore.rotation.x += dt * 0.1;

              d.forcefield.material.uniforms.uTime.value = time;
              d.forcefield.material.uniforms.uActivation.value =
                d.activationLevel;

              const ffScale = lerp(1.0, 1.2, d.activationLevel);
              d.forcefield.scale.setScalar(ffScale);
              d.forcefield.rotation.y -= dt * (0.1 + d.activationLevel * 2.0);
            }
          });

          innerAgents.forEach((agent) => {
            const d = agent.userData;
            const targetY = d.basePos.y + Math.sin(time + d.id) * 1.5;
            agent.position.x = lerp(agent.position.x, d.basePos.x, dt * 2.0);
            agent.position.z = lerp(agent.position.z, d.basePos.z, dt * 2.0);
            agent.position.y = lerp(agent.position.y, targetY, dt * 2.0);

            const targetActivation = d.isActive ? 1.0 : 0.0;
            const lerpFactor = 1.0 - Math.exp(-dt * 10.0);
            d.activationLevel +=
              (targetActivation - d.activationLevel) * lerpFactor;

            const pop =
              Math.sin(d.activationLevel * Math.PI) *
              Math.pow(Math.sin(d.activationLevel * Math.PI), 0.5) *
              1.5;
            agent.rotation.y += dt * (0.2 + pop * 10.0);

            const activation = d.activationLevel;
            const activeScale = 1.0 + activation * 0.5 + pop * 0.5;

            if (d.auraMat) {
              d.auraMat.uniforms.uTime.value = time;
              d.auraMat.uniforms.uActivation.value = activation;
            }

            if (d.isLegal) {
              d.pillar.rotation.y = time * 0.5 + activation * 10.0;
              d.pillar.material.emissiveIntensity = 1.0 + activation * 8.0;
              d.pillar.scale.setScalar(1.0 + pop * 0.5);

              const bounce = Math.sin(time * 2.0) * 0.3;
              const balance = bounce * (1.0 - activation);

              d.scales.forEach((sc, i) => {
                sc.angle += dt * (1.0 + activation * 4.0);
                sc.mesh.position.x = Math.cos(sc.angle) * 2.0;
                sc.mesh.position.z = Math.sin(sc.angle) * 2.0;
                const yOffset = i === 0 ? balance : -balance;
                sc.mesh.position.y =
                  Math.sin(time * 3.0 + i * Math.PI) * 0.2 + yOffset;
                sc.mesh.rotation.y -= dt;
                sc.mesh.rotation.z = Math.sin(time * 2.0 + i) * 0.2;
                sc.mesh.material.emissiveIntensity = 1.0 + activation * 3.0;
              });

              d.orbit.rotation.z = time * 0.5 + activation * 5.0;
              d.orbit.scale.setScalar(activeScale);
            } else if (d.isValidation) {
              d.hexCore.rotation.y = time * 0.2 + activation * 10.0;
              d.hexCore.material.emissiveIntensity = 1.0 + activation * 8.0;
              d.hexCore.scale.setScalar(1.0 + pop);

              d.scanners.forEach((sc) => {
                const phase =
                  (time * (1.0 + activation * 5.0) + sc.offset) % 3.0;
                const yPos = phase - 1.5;
                sc.mesh.position.y = yPos;
                sc.mesh.rotation.y += dt * 4.0;
                sc.mesh.scale.setScalar(1.0 + activation * 0.5 + pop);
                sc.mesh.material.opacity = 0.5 + activation * 0.5;
              });
            }
          });

          // Inter link logic
          if (innerAgents.length === 2 && interLinkMesh) {
            const p1 = innerAgents[0].position;
            const p2 = innerAgents[1].position;
            const dist = p1.distanceTo(p2);

            interLinkMesh.position.copy(p1);
            interLinkMesh.lookAt(p2);

            interLinkMat.uniforms.uTime.value = time;

            const maxAct = Math.max(
              innerAgents[0].userData.activationLevel,
              innerAgents[1].userData.activationLevel,
            );
            const baseScale = 1.0 + maxAct * 2.0;

            interLinkMesh.scale.set(baseScale, baseScale, dist);
          }

          const targetPkgY =
            packagerGroup.userData.basePos.y + Math.sin(time) * 1.5;
          packagerGroup.position.x = lerp(
            packagerGroup.position.x,
            packagerGroup.userData.basePos.x,
            dt * 2.0,
          );
          packagerGroup.position.z = lerp(
            packagerGroup.position.z,
            packagerGroup.userData.basePos.z,
            dt * 2.0,
          );
          packagerGroup.position.y = lerp(
            packagerGroup.position.y,
            targetPkgY,
            dt * 2.0,
          );

          packagerGroup.userData.pulse = lerp(
            packagerGroup.userData.pulse,
            0.0,
            dt * 6.0,
          );
          packagerGroup.userData.coreMat.uniforms.uTime.value = time;
          packagerGroup.userData.coreMat.uniforms.uPulse.value = lerp(
            packagerGroup.userData.coreMat.uniforms.uPulse.value,
            0.0,
            dt * 8.0,
          );

          // Re-scale cage based on pulse
          const cageScale = 1.0 + packagerGroup.userData.pulse * 0.8;
          packagerGroup.userData.cage.scale.setScalar(cageScale);
          packagerGroup.userData.cage.rotation.y =
            time * 0.4 + packagerGroup.userData.pulse * 2.0;
          packagerGroup.userData.cage.rotation.z =
            time * 0.2 + packagerGroup.userData.pulse * 2.0;

          // Rotate rings
          const pulseFactor = 1.0 + packagerGroup.userData.pulse * 15.0;
          packagerGroup.userData.rings.forEach((ring, idx) => {
            ring.rotation.x += dt * (0.5 + idx * 0.2) * pulseFactor;
            ring.rotation.y += dt * (0.3 + idx * 0.4) * pulseFactor;
            ring.scale.setScalar(1.0 + packagerGroup.userData.pulse * 0.5);
          });

          for (let i = dataStreams.length - 1; i >= 0; i--) {
            const stream = dataStreams[i];
            if (stream.delay > 0) {
              stream.delay -= dt;
              continue;
            }
            if (!stream.active) {
              stream.active = true;
            }

            stream.progress += dt * 0.15;

            if (stream.progress >= 1.0 && !stream.hasFired) {
              stream.hasFired = true;
              if (stream.onComplete) stream.onComplete();
            }

            stream.material.uniforms.uTime.value = time;
            stream.material.uniforms.uFlowProgress.value = stream.progress;

            const sP = stream.startNode.position;
            const eP = stream.endNode.position;

            stream.material.uniforms.uStart.value.copy(sP);
            stream.material.uniforms.uEnd.value.copy(eP);

            _tempDir.subVectors(eP, sP).multiplyScalar(0.2);
            stream.material.uniforms.uControl1.value
              .copy(sP)
              .add(_upOffset)
              .add(_tempDir);
            stream.material.uniforms.uControl2.value
              .copy(eP)
              .add(_downOffset)
              .sub(_tempDir);

            if (stream.progress >= 1.8) {
              stream.mesh.visible = false;
              dataStreamPool.push({
                mesh: stream.mesh,
                material: stream.material,
              });
              dataStreams.splice(i, 1);
            }
          }

          renderer.render(scene, camera);
        }

        window.addEventListener("resize", () => {
          camera.aspect = window.innerWidth / window.innerHeight;
          camera.updateProjectionMatrix();
          renderer.setSize(window.innerWidth, window.innerHeight);
        });

        animate();
      } catch (err) {
        document.body.innerHTML +=
          '<div style="position:absolute;top:0;left:0;z-index:99999;color:red;background:white;padding:20px;font-family:monospace;white-space:pre-wrap;">' +
          err.stack +
          "</div>";
        console.error(err);
      }
    </script>
  </body>
</html>

        function createTallPlant() {
          const plantContainer = new THREE.Group();

          // Elegant tall matte ceramic pot with a bit of a curve

          const potMat = new THREE.MeshStandardMaterial({

            roughness: 1.0,

          const soil = new THREE.Mesh(soilGeom, soilMat);
          soil.position.y = 21.5;
          plantContainer.add(soil);

          // Lush Parametric Palm (Areca / Kentia Palm style)
          // We will create several sweeping fronds emitting from the center.
          const frondCount = 28; // very lush
          const leafColor1 = new THREE.Color(0x284022); // Deep green
          const leafColor2 = new THREE.Color(0x4a7a37); // Lighter green
          const stemMat = new THREE.MeshStandardMaterial({
            color: 0x3d5c2c,
            roughness: 0.6,
          });

          // Pre-create leaflet geometries to save memory
          // A leaflet is an elongated curved shape. We will use a scaled sphere or box.
          // Or simply a BoxBufferGeometry rotated, which looks good if dense.
          const leafletGeom = new THREE.BoxBufferGeometry(0.3, 0.05, 1);
          leafletGeom.translate(0, 0, 0.5); // Pivot at the base

          for (let i = 0; i < frondCount; i++) {
            const frondGroup = new THREE.Group();

            // Randomize frond parameters
            const angle =
              (i / frondCount) * Math.PI * 2 + (Math.random() - 0.5) * 0.4;
            const length = 40 + Math.random() * 35; // taller! up to 75 units
            const arc = 0.3 + Math.random() * 0.5; // how far it arcs outward (0 to 1)

            // We use a curve for the main stem
            const startPt = new THREE.Vector3(
              (Math.random() - 0.5) * 3,
              21.5,
              (Math.random() - 0.5) * 3,
            );
            const midPt = new THREE.Vector3(
              Math.cos(angle) * (length * arc * 0.3),
              21.5 + length * 0.6,
              Math.sin(angle) * (length * arc * 0.3),
            );
            const endPt = new THREE.Vector3(
              Math.cos(angle) * (length * arc),
              21.5 + length * (1.0 - arc * 0.4),
              Math.sin(angle) * (length * arc),
            );

            const stemCurve = new THREE.QuadraticBezierCurve3(
              startPt,
              midPt,
              endPt,
            );
            const sGeom = new THREE.TubeBufferGeometry(
              stemCurve,
              12,
              0.25,
              6,
              false,
            );
            const stem = new THREE.Mesh(sGeom, stemMat);
            frondGroup.add(stem);

            // Add leaflets along the stem
            const leafletMat = new THREE.MeshStandardMaterial({
              color: leafColor1
                .clone()
                .lerp(leafColor2, Math.random() * 0.6 + 0.2),
              roughness: 0.5,
              metalness: 0.1,
              side: THREE.DoubleSide,
            });

            // Start leaflets partway up the stem
            const numLeaflets = Math.floor(length * 1.5);
            for (let j = 0; j < numLeaflets; j++) {
              const t = 0.2 + (0.8 * j) / numLeaflets; // from 20% to 100% of the stem
              const pos = stemCurve.getPoint(t);
              const tangent = stemCurve.getTangent(t).normalize();

              // We need a normal vector to point the leaflets horizontally outward from the stem
              // A simple way is to cross the tangent with an "up" vector, though tangent itself might point up.
              // Let's create a local basis.
              const up = new THREE.Vector3(0, 1, 0);
              let right = new THREE.Vector3().crossVectors(tangent, up);
              if (right.lengthSq() < 0.001) {
                // If tangent is completely vertical, just pick X
                right.set(1, 0, 0);
              }
              right.normalize();

              // The leaflets on the left and right of the stem
              for (let side = -1; side <= 1; side += 2) {
                const leaflet = new THREE.Mesh(leafletGeom, leafletMat);
                leaflet.position.copy(pos);

                // Orient the leaflet
                // Point it slightly along the stem (tangent), mostly outwards (right * side)
                // and droop slightly downwards (-up)
                const droop = (1.0 - t) * 0.5 + 0.2; // upper leaflets droop more/less? Let's say droop increases towards tips
                const dir = new THREE.Vector3()
                  .copy(right)
                  .multiplyScalar(side * 1.5)
                  .add(tangent.clone().multiplyScalar(1.2))
                  .add(up.clone().multiplyScalar(-droop))
                  .normalize();

                // Use lookAt to reorient the leaflet
                // Since leaflet base is at origin, we create a dummy object to use lookAt
                const target = pos.clone().add(dir);
                leaflet.lookAt(target);

                // Scale the leaflet (longer near the middle of the leafy section, tapering at the tip)
                // t is 0.2 to 1.0. Let's map it to a parabolic scale.
                const leafT = (t - 0.2) / 0.8; // 0 to 1
                const lengthScale = Math.sin(leafT * Math.PI) * 4.0 + 1.0;
                leaflet.scale.set(1, 1, lengthScale * (length / 50));

                frondGroup.add(leaflet);
              }
            }

            plantContainer.add(frondGroup);
          }

          return plantContainer;
        }

          string1.rotation.x = -0.45; // Angulada hacia la barra superior
          pendulum.add(string1);

          const string2 = new THREE.Mesh(nStringGeom, stringMat);
          string2.position.set(0, -5.0, -2.25);
          string2.rotation.x = 0.45;
          pendulum.add(string2);

          newtonGroup.add(pendulum);

        // === 5. Planta Paramétrica (Bonsái Suculenta en Maceta) ===
        const plantGroup = new THREE.Group();
        // Ocupando el espacio libre dejado por la lámpara
        plantGroup.position.set(-78, -8, -65);
        deskGroup.add(plantGroup);

        // Maceta de Cerámica Mate
        const potGeom = new THREE.CylinderBufferGeometry(5, 3.5, 6, 64);
        const potMat = new THREE.MeshStandardMaterial({
          color: 0xeaeaea,
          roughness: 0.3,
          metalness: 0.1,
        });
        const pot = new THREE.Mesh(potGeom, potMat);
        pot.position.y = 3;
        plantGroup.add(pot);

        // Tierra Sintética
        const soilGeom = new THREE.CylinderBufferGeometry(4.8, 4.8, 0.2, 64);
        const soilMat = new THREE.MeshStandardMaterial({
          color: 0x221105,
          roughness: 1.0,
        });
        const soil = new THREE.Mesh(soilGeom, soilMat);
        soil.position.y = 5.8;
        plantGroup.add(soil);

        // Generación Matemática de Hojas de Suculenta (Espiral de Fibonacci)
        const leafColor1 = new THREE.Color(0x2d4c1e); // Verde oscuro
        const leafColor2 = new THREE.Color(0x76b041); // Verde brillante
        const leafGeom = new THREE.SphereBufferGeometry(1, 32, 32);

        const leafCount = 60;
        const goldenAngle = Math.PI * (3 - Math.sqrt(5)); // ~137.5 grados

        for (let i = 0; i < leafCount; i++) {
          const radius = Math.sqrt(i) * 0.45;
          const theta = i * goldenAngle;

          const leafMat = new THREE.MeshStandardMaterial({
            color: leafColor1.clone().lerp(leafColor2, i / leafCount),
            roughness: 0.5,
            metalness: 0.05,
          });

          const leaf = new THREE.Mesh(leafGeom, leafMat);

          // Compresión y estiramiento para fomar la hoja carnosa
          const size = 0.4 + (i / leafCount) * 0.9;
          leaf.scale.set(0.7 * size, 0.25 * size, 2.2 * size);

          const x = radius * Math.cos(theta);
          const z = radius * Math.sin(theta);
          // Caída orgánica dependiendo del radio
          const y =
            6.0 + (1 - i / leafCount) * 1.5 - Math.pow(radius, 2) * 0.12;

          leaf.position.set(x, y, z);
          leaf.lookAt(new THREE.Vector3(x * 2, y + 5, z * 2));

          plantGroup.add(leaf);
        }
